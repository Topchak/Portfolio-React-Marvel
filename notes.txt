import ErrorMessage from "../ErrorMessage/ErrorMessage";

class ErrorBoundary extends Component {

  state = {
    error: false
  }

  componentDidCatch( error, errorInfo ) {             // create stub errors
    this.setState({
      error: true
    })
  }

  render (){ 
    if ( this.state.error ) {
      return <ErrorMessage/>
    }
    return this.props.children
  }
}

export default ErrorBoundary
